#include <stdio.h>
#include <string.h>

int main(int argc, char *argv[]) {
    if (argc != 2) {
        printf("Usage: %s <password>\n", argv[0]);
        return 1;
    }

    // كلمة المرور هي "Pwn!" ولكنها مخزنة كقيمة عددية 
    // بالـ Hex: 0x216e7750 (تخزن عكسياً Little-Endian)
    int secret = 0x216e7750; 
    
    int input;
    memcpy(&input, argv[1], 4);

    if (input == secret) {
        printf("Access Granted! Flag: pwn{L1ttl3_End1an_M4st3r}\n");
    } else {
        printf("Access Denied!\n");
    }
    return 0;
}
